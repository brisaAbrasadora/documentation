Objects

String value

When an object is converted to a string, or printed directly, the toString() method inherited from the class Object is called. By default, it prints "[object Object]", but we can override the method on our class. 

// at person.class.js
class Person {
    #name;
    #age;

    constructor(name, age) {
        this.#name = name;
        this.#age = age;
    };

    toString() {
        return "My name is " . this.#name . " and I am " . this.#age;
    };
};

//at index.js
let person1 = new Person('Sabrina', 21);

console.log(person1);

My name is Sabrina and I am 21

Primitive value

When comparing objects using relational operators, the primitive value of the object is obtained and compared using the valueOf() method inherited from the class Object. 

// at person.class.js
class Person {
    #name;
    #age;

    constructor(name, age) {
        this.#name = name;
        this.#age = age;
    };

    toString() {
        return this.#name;
    }

    valueOf() {
        return this.#age;
    }
}

//at index.js
let person1 = new Person('Sabrina', 21);
let person2 = new Person('Dani', 23)

if (person1 > person2) {
    console.log(person1 . "is older than " . person2);
} else {
    console.log(person2 . "is older than " . person1);
};

Dani is older than Sabrina

In case valueOf() is not declared, the toString() value will be used to compare.

Destructuring objects

We can destructure object properties. This means we can set up an object and declare its properties easily.
We set an object that serves as a template. Later on, we declare two variables that matches the object attributes and asign the values from the template. We can now access to those values from another variable, keeping us safe from altering the original value. We can access to the data more efficiently.

In this example, we have an object with three attributes. Then, we will create the variables name and age, which are names of the attributes we want to access to. The id attribute is ignored.

let user1 = {
    id: 1,
    name: 'Vivi',
    age: 18,
};

const {name, age} = user1;

console.log(`My name is ${name} and I am ${age}`);

My name is Vivi and I am 18

It is required to know the exact name of the attribute we want to access, but we can rename it for the part of the code we are working with and access to the attributes from different objects, since in the previous example the variables name and age are declared and no longer modifiable. Here is shown that an object created with a class can also be destructured.

class User {
    id;
    name;
    age;

    constructor(id, name, age) {
        this.id = id;
        this.name = name;
        this.age = age;
    };
};

let user1 = {
    id: 1,
    name: 'Vivi',
    age: 18,
};

let user2 = new User(2, 'Didi', 23);

const {name: nickname, age: howOld} = user1;

console.log(`My name is ${nickname} and I am ${howOld}`);

const {name: nickname2, age: howOld2} = user2;

console.log(`My name is ${nickname2} and I am ${howOld2}`);

My name is Vivi and I am 18
My name is Didi and I am 23

An object can also be destructured on a function!

function printUserData({name, age}, otherInfo = "None") {
    console.log('User name: ', name);
    console.log('User age: ', age);
    console.log('Other info: ', otherInfo);
}

let user1 = {
    id: 1,
    name: 'Vivi',
    age: 18,
};


printUserData(user1);

User name: Vivi
User age: 18
Other info: None

Combining objects

We can use object destructuration to create new objects combining properties from 2 or more objects. When a property is present in 2 or more objects, the last object's value will remain.

function configGame(options) {
    let defaults = {
        name: 'Player 1',
        level: 1,
        difficulty: 'normal',
        genre: 'male',
    };

    let config = {...defaults, ...options};
    console.log(config);
};

let options = {
    name: 'Vivianne Gorch',
    genre: 'female',
};

configGame(options);
{
    name: 'Vivianne Gorch',
    level: 1,
    difficulty: 'normal',
    genre: 'female'
}
